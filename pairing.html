// pairingserver.js import express from 'express'; import cors from 'cors'; import fs from 'fs'; import path from 'path'; import { fileURLToPath } from 'url'; import { requestPairingCode } from './utils/connector.js'; const __filename = fileURLToPath(import.meta.url); const __dirname = path.dirname(__filename); const app = express(); const port = process.env.PORT || 19112; // ← modifié ici const sessionsDir = path.join(__dirname, 'sessions'); if (!fs.existsSync(sessionsDir)) fs.mkdirSync(sessionsDir, { recursive: true }); app.use(cors()); app.use(express.json()); app.get('/pair', async (req, res) => { const number = req.query.number; if (!number || !/^\d{8,15}$/.test(number)) { return res.status(400).json({ error: 'Numéro invalide (8 à 15 chiffres)' }); } try { const code = await requestPairingCode(number); res.json({ code }); } catch (err) { console.error('Erreur pairing:', err); res.status(500).json({ error: err.message || 'Échec de génération du code' }); } }); app.get('/sessions/count', (req, res) => { try { if (!fs.existsSync(sessionsDir)) return res.json({ count: 0 }); const folders = fs.readdirSync(sessionsDir); const count = folders.filter(f => f.startsWith('session_')).length; res.json({ count }); } catch(e) { res.json({ count: 0 }); } }); app.get('/health', (req, res) => { res.json({ status: 'ok', timestamp: Date.now() }); }); export function startPairingServer() { app.listen(port, '0.0.0.0', () => { console.log(`🔗 Serveur de pairing actif sur http://0.0.0.0:${port}`); }); }
